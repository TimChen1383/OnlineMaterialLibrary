    // Platform variables: uv, normal, position, time, resolution
    vec2 fragCoord = uv * resolution;
    vec2 uv_local = (2.0 * fragCoord - resolution) / min(resolution.x, resolution.y);

    for(float i = 1.0; i < 10.0; i++) {
        uv_local.x += 0.6 / i * cos(i * 2.5 * uv_local.y + time);
        uv_local.y += 0.6 / i * cos(i * 1.5 * uv_local.x + time);
    }

    finalColor = vec3(0.1) / abs(sin(time - uv_local.y - uv_local.x));
